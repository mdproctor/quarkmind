package io.quarkmind.sc2.emulated;

public sealed interface AttackStrategy
    permits AttackStrategy.Threshold, AttackStrategy.Timer, AttackStrategy.Hybrid {

    record Threshold(int minArmy) implements AttackStrategy {}
    record Timer(int everyNTicks) implements AttackStrategy {}
    record Hybrid(int minArmy, int everyNTicks) implements AttackStrategy {}

    default boolean shouldAttack(int armySize, long gameFrame) {
        return switch (this) {
            case Threshold(int min)         -> armySize >= min;
            case Timer(int every)           -> gameFrame > 0 && gameFrame % every == 0;
            case Hybrid(int min, int every) -> armySize >= min && gameFrame > 0 && gameFrame % every == 0;
        };
    }
}
```

package io.quarkmind.sc2.emulated;

import io.quarkmind.domain.*;
import org.jboss.logging.Logger;
import java.util.*;
import java.util.function.Supplier;

/**
 * Active enemy AI. Owns the enemy economic loop (mineral accumulation, build order,
 * unit production) and attack decision. Not a CDI bean — owned by {@link EmulatedGame}.
 *
 * Holds references to EmulatedGame's shared lists (enemyUnits, enemyBuildings,
 * pendingCompletions, enemyTargets) and writes to them each tick. All access is
 * single-threaded (called from EmulatedGame.tick()).
 */
class EnemyAI {

    private static final Logger log = Logger.getLogger(EnemyAI.class);
    private static final int MAX_WORKERS = 8;
    private static final Point2d SPAWN_POS = new Point2d(26, 26);
    private static final Point2d NEXUS_POS = new Point2d(8, 8);

    private final EnemyRace race;
    private final AttackStrategy strategy;
    private final List<Unit> enemyUnits;
    private final List<Building> enemyBuildings;
    private final List<PendingCompletion> pendingCompletions;
    private final Map<String, Point2d> enemyTargets;
    private final Supplier<String> tagSupplier;

    private double mineralAccumulator;
    private int supply;
    private int supplyUsed;
    private boolean buildingSupply = false;
    private boolean buildingProduction = false;

    EnemyAI(EnemyRace race, AttackStrategy strategy,
            List<Unit> enemyUnits, List<Building> enemyBuildings,
            List<PendingCompletion> pendingCompletions,
            Map<String, Point2d> enemyTargets,
            Supplier<String> tagSupplier) {
        this.race               = race;
        this.strategy           = strategy;
        this.enemyUnits         = enemyUnits;
        this.enemyBuildings     = enemyBuildings;
        this.pendingCompletions = pendingCompletions;
        this.enemyTargets       = enemyTargets;
        this.tagSupplier        = tagSupplier;
    }

    void reset() {
        mineralAccumulator = SC2Data.INITIAL_MINERALS;
        supply             = SC2Data.supplyBonus(race.mainBuilding); // 11 for COMMAND_CENTER
        supplyUsed         = 0;
        buildingSupply     = false;
        buildingProduction = false;

        // Initial main building (complete from the start)
        String ccTag = tagSupplier.get();
        int ccHp = SC2Data.maxBuildingHealth(race.mainBuilding);
        enemyBuildings.add(new Building(ccTag, race.mainBuilding, SPAWN_POS, ccHp, ccHp, true));

        // Initial workers
        for (int i = 0; i < 6; i++) {
            String tag = tagSupplier.get();
            int hp = SC2Data.maxHealth(race.workerType);
            enemyUnits.add(new Unit(tag, race.workerType,
                new Point2d(SPAWN_POS.x() + i * 0.5f, SPAWN_POS.y()),
                hp, hp, SC2Data.maxShields(race.workerType), SC2Data.maxShields(race.workerType)));
            supplyUsed += SC2Data.supplyCost(race.workerType);
        }
        log.debugf("[ENEMY AI] Reset — race=%s supply=%d/%d", race, supplyUsed, supply);
    }

    void tick(long gameFrame) {
        long workerCount = enemyUnits.stream()
            .filter(u -> u.type() == race.workerType).count();
        mineralAccumulator += workerCount * SC2Data.MINERALS_PER_PROBE_PER_TICK;

        // Build order — priority, first match wins (one action per tick)
        if (supplyBlocked() && !buildingSupply
                && (int) mineralAccumulator >= SC2Data.mineralCost(race.supplyBuilding)) {
            buildSupply(gameFrame);
        } else if (!hasCompleted(race.productionBuilding) && !buildingProduction
                && (int) mineralAccumulator >= SC2Data.mineralCost(race.productionBuilding)) {
            buildProduction(gameFrame);
        } else if (workerCount < MAX_WORKERS
                && (int) mineralAccumulator >= SC2Data.mineralCost(race.workerType)
                && supplyUsed + SC2Data.supplyCost(race.workerType) <= supply) {
            trainWorker(gameFrame);
        } else if (hasCompleted(race.productionBuilding)
                && (int) mineralAccumulator >= SC2Data.mineralCost(combatUnitType())
                && supplyUsed + SC2Data.supplyCost(combatUnitType()) <= supply) {
            trainCombatUnit(gameFrame);
        }

        // Attack decision
        long armySize = enemyUnits.stream()
            .filter(u -> u.type() != race.workerType).count();
        if (strategy.shouldAttack((int) armySize, gameFrame)) {
            enemyUnits.stream()
                .filter(u -> u.type() != race.workerType)
                .forEach(u -> enemyTargets.put(u.tag(), NEXUS_POS));
            if (armySize > 0) log.infof("[ENEMY AI] Attack! army=%d frame=%d", armySize, gameFrame);
        }
    }

    private boolean supplyBlocked() {
        return supplyUsed >= supply - 2;
    }

    private boolean hasCompleted(BuildingType type) {
        return enemyBuildings.stream().anyMatch(b -> b.type() == type && b.isComplete());
    }

    private UnitType combatUnitType() {
        return race.combatUnits.get(0);
    }

    private void buildSupply(long gameFrame) {
        mineralAccumulator -= SC2Data.mineralCost(race.supplyBuilding);
        buildingSupply = true;
        String tag = tagSupplier.get();
        int hp = SC2Data.maxBuildingHealth(race.supplyBuilding);
        Point2d pos = new Point2d(SPAWN_POS.x() + 3, SPAWN_POS.y());
        enemyBuildings.add(new Building(tag, race.supplyBuilding, pos, hp, hp, false));
        long completesAt = gameFrame + SC2Data.buildTimeInTicks(race.supplyBuilding);
        pendingCompletions.add(new PendingCompletion(completesAt, () -> {
            markComplete(tag);
            supply += SC2Data.supplyBonus(race.supplyBuilding);
            buildingSupply = false;
            log.debugf("[ENEMY AI] Supply building complete (supply now %d)", supply);
        }));
        log.debugf("[ENEMY AI] Building %s (tag=%s)", race.supplyBuilding, tag);
    }

    private void buildProduction(long gameFrame) {
        mineralAccumulator -= SC2Data.mineralCost(race.productionBuilding);
        buildingProduction = true;
        String tag = tagSupplier.get();
        int hp = SC2Data.maxBuildingHealth(race.productionBuilding);
        Point2d pos = new Point2d(SPAWN_POS.x() + 5, SPAWN_POS.y());
        enemyBuildings.add(new Building(tag, race.productionBuilding, pos, hp, hp, false));
        long completesAt = gameFrame + SC2Data.buildTimeInTicks(race.productionBuilding);
        pendingCompletions.add(new PendingCompletion(completesAt, () -> {
            markComplete(tag);
            buildingProduction = false;
            log.debugf("[ENEMY AI] Production building complete");
        }));
        log.debugf("[ENEMY AI] Building %s (tag=%s)", race.productionBuilding, tag);
    }

    private void trainWorker(long gameFrame) {
        mineralAccumulator -= SC2Data.mineralCost(race.workerType);
        supplyUsed += SC2Data.supplyCost(race.workerType); // reserve at queue time
        long completesAt = gameFrame + SC2Data.trainTimeInTicks(race.workerType);
        pendingCompletions.add(new PendingCompletion(completesAt, () -> {
            String tag = tagSupplier.get();
            int hp = SC2Data.maxHealth(race.workerType);
            enemyUnits.add(new Unit(tag, race.workerType, SPAWN_POS, hp, hp,
                SC2Data.maxShields(race.workerType), SC2Data.maxShields(race.workerType)));
            log.debugf("[ENEMY AI] Trained %s (tag=%s)", race.workerType, tag);
        }));
    }

    private void trainCombatUnit(long gameFrame) {
        UnitType type = combatUnitType();
        mineralAccumulator -= SC2Data.mineralCost(type);
        supplyUsed += SC2Data.supplyCost(type); // reserve at queue time
        long completesAt = gameFrame + SC2Data.trainTimeInTicks(type);
        pendingCompletions.add(new PendingCompletion(completesAt, () -> {
            String tag = tagSupplier.get();
            int hp = SC2Data.maxHealth(type);
            enemyUnits.add(new Unit(tag, type, SPAWN_POS, hp, hp,
                SC2Data.maxShields(type), SC2Data.maxShields(type)));
            log.debugf("[ENEMY AI] Trained %s (tag=%s)", type, tag);
        }));
    }

    private void markComplete(String tag) {
        enemyBuildings.replaceAll(b -> b.tag().equals(tag)
            ? new Building(b.tag(), b.type(), b.position(), b.health(), b.maxHealth(), true)
            : b);
    }
}
```
